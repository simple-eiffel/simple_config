	SIMPLE_CONFIG

feature -- Access

	file_path: STRING_8
			-- Path to configuration file

	is_modified: BOOLEAN
			-- Has configuration been modified since last save?

	has_key (a_key: STRING_8): BOOLEAN
			-- Does configuration have a_key?
			-- Supports dot notation: "database.host"
		require
			key_not_empty: not a_key.is_empty
	
feature -- String access

	string_value (a_key: STRING_8): detachable STRING_8
			-- Get string value for a_key.
			-- Supports dot notation: "database.host"
		require
			key_not_empty: not a_key.is_empty

	string_value_or_default (a_key, a_default: STRING_8): STRING_8
			-- Get string value for a_key, or a_default if not found.
		require
			key_not_empty: not a_key.is_empty

	string_value_or_env (a_key, a_env_var: STRING_8): detachable STRING_8
			-- Get string value for a_key, or environment variable a_env_var.
		require
			key_not_empty: not a_key.is_empty
			env_var_not_empty: not a_env_var.is_empty

	string_value_or_env_or_default (a_key, a_env_var, a_default: STRING_8): STRING_8
			-- Get string value for a_key, or env var a_env_var, or a_default.
		require
			key_not_empty: not a_key.is_empty
			env_var_not_empty: not a_env_var.is_empty
	
feature -- Integer access

	integer_value (a_key: STRING_8): INTEGER_32
			-- Get integer value for a_key (0 if not found).
			-- Supports dot notation.
		require
			key_not_empty: not a_key.is_empty

	integer_value_or_default (a_key: STRING_8; a_default: INTEGER_32): INTEGER_32

	integer_value_or_env (a_key, a_env_var: STRING_8; a_default: INTEGER_32): INTEGER_32
			-- Get integer from config, or parse from env var, or use default.
		require
			key_not_empty: not a_key.is_empty
			env_var_not_empty: not a_env_var.is_empty

	boolean_value (a_key: STRING_8): BOOLEAN
			-- Get boolean value for a_key (False if not found).
			-- Supports dot notation.

	boolean_value_or_default (a_key: STRING_8; a_default: BOOLEAN): BOOLEAN

	real_value (a_key: STRING_8): REAL_64

	real_value_or_default (a_key: STRING_8; a_default: REAL_64): REAL_64
			-- Get real value for a_key, or a_default if not found.
		require
			key_not_empty: not a_key.is_empty
	
feature -- Array access

	string_list (a_key: STRING_8): ARRAYED_LIST [STRING_8]
			-- Get array of strings for a_key.
			-- Supports dot notation.

	integer_list (a_key: STRING_8): ARRAYED_LIST [INTEGER_32]

	section (a_key: STRING_8): detachable SIMPLE_CONFIG

			-- Supports dot notation.

feature -- Element change

	set_string (a_key, a_value: STRING_8)
			-- Set string value for a_key.
			-- Note: Does NOT support dot notation for setting (use sections).
		require
			key_not_empty: not a_key.is_empty
		ensure
			modified: is_modified

	set_integer (a_key: STRING_8; a_value: INTEGER_32)
			-- Set integer value for a_key.
		require
			key_not_empty: not a_key.is_empty
		ensure
			modified: is_modified

	set_boolean (a_key: STRING_8; a_value: BOOLEAN)
			-- Set boolean value for a_key.
		require
			key_not_empty: not a_key.is_empty
		ensure
			modified: is_modified

	set_real (a_key: STRING_8; a_value: REAL_64)
			-- Set real value for a_key.
		require
			key_not_empty: not a_key.is_empty
		ensure
			modified: is_modified

	set_section (a_key: STRING_8; a_section: SIMPLE_CONFIG)
			-- Set nested section for a_key.
		require
			key_not_empty: not a_key.is_empty
			section_not_void: a_section /= Void
		ensure
			modified: is_modified

	remove (a_key: STRING_8)
			-- Remove value for a_key.
		require
			key_not_empty: not a_key.is_empty
		ensure
			modified: is_modified
	
feature -- File operations

	load
			-- Load configuration from file.

	merge_file (a_file_path: STRING_8)
			-- Merge another config file, overriding existing values.
		require
			path_not_empty: not a_file_path.is_empty

	save
			-- Save configuration to file.
		require
			has_file_path: not file_path.is_empty
		ensure
			not_modified: not is_modified

	save_to (a_file_path: STRING_8)
			-- Save configuration to specified file.
		require
			path_not_empty: not a_file_path.is_empty
		ensure
			path_set: file_path = a_file_path
			not_modified: not is_modified

	to_json: STRING_8
			-- Convert configuration to compact JSON string.

	to_json_pretty: STRING_8
			-- Convert configuration to formatted JSON string.
